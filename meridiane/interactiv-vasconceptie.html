// Script de securitate pentru a bloca copierea și uneltele de dezvoltator document.addEventListener('contextmenu', event => event.preventDefault()); // Blochează clicul dreapta document.addEventListener('keydown', function(e) { // Blochează Ctrl+C (Copiere) if (e.ctrlKey && e.key === 'c') { e.preventDefault(); } // Blochează Ctrl+U (View Source) if (e.ctrlKey && e.key === 'u') { e.preventDefault(); } // Blochează F12 (Developer Tools) if (e.key === 'F12') { e.preventDefault(); } // Blochează Ctrl+Shift+I (Developer Tools) if (e.ctrlKey && e.shiftKey && e.key === 'I') { e.preventDefault(); } // Blochează Ctrl+Shift+J (Developer Tools Console) if (e.ctrlKey && e.shiftKey && e.key === 'J') { e.preventDefault(); } });
